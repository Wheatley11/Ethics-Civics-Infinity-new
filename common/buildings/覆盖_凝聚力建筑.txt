#媒体大楼
building_bureaucratic_1 = {
	base_buildtime = @b1_time
	icon = building_bureaucratic_1
	category = unity

	potential = {
		exists = owner
		allow_bureaucratic_building = yes
	}

	allow = {
		OR = {
			is_adminless_country = no
			is_clean_government_country = yes
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			allow_bureaucratic_building = no
			AND = {
				is_adminless_country = yes
				is_clean_government_country = no
			}
		}
	}

	convert_to = {
		building_temple
		building_hive_node
		building_uplink_node
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	planet_modifier = {
		job_media_worker_add = @b1_jobs
	}

	triggered_desc = {
		text = job_media_worker_effect_desc
	}

	upgrades = {
		"building_bureaucratic_2"
	}

	prerequisites = {
		"tech_planetary_government"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
	}
	ai_resource_production = {
		unity = 20
	}
}

building_autochthon_monument = {
	base_buildtime = @b1_time
	base_cap_amount = 1
	category = unity

	potential = {
		exists = owner
		NOR = {
			has_modifier = slave_colony
			has_planet_flag = planetary_station
		}
		owner = {
			NOR = {
				is_gestalt = yes
				OR = {
					has_authority = auth_corporate
					has_civic = civic_galactic_sovereign_megacorp

					has_civic = civic_world_cooperative
					has_valid_civic = civic_triple_the_profit
				}
				is_memorialist_empire = yes
			}
		}
	}

	destroy_trigger = {
		exists = owner
		OR = {
			has_modifier = slave_colony
			owner = {
				OR = {
					is_gestalt = yes
					OR = {
					has_authority = auth_corporate
					has_civic = civic_galactic_sovereign_megacorp

					has_civic = civic_world_cooperative
					has_valid_civic = civic_triple_the_profit
				}
					is_memorialist_empire = yes
				}
			}
		}
	}

	resources = {
		category = planet_buildings
		produces = {
			trigger = { exists = owner }
			unity = 1
			multiplier = owner.trigger:num_ascension_perks
		}
		produces = {
			trigger = {
				exists = owner
				owner = {
					has_valid_civic = civic_futuristic_art
				}
			}
			physics_research = 0.5
			engineering_research = 0.5
			society_research = 0.5

		}

			minerals = @b1_minerals
		}
	}

	}

	triggered_planet_modifier = {

			exists = owner 
			owner = {
				has_valid_civic = civic_sparta_spirit
			}
		}
		modifier = {
			job_culture_worker_add = -2
			job_commandante_add = 2
		}
	}


	triggered_desc = {
		text = building_autochthon_monument_effect
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				has_valid_civic = civic_futuristic_art
			}
		}
		text = building_art_club_effect
	}

	triggered_desc = {
		trigger = {
			exists = owner 
			owner = {
				NOT = {
					has_valid_civic = civic_sparta_spirit
				}
			}
		}
		text = job_culture_worker_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner 
			owner = {
				has_valid_civic = civic_sparta_spirit
			}
		}
		text = job_commandante_effect_desc
	}

	prerequisites = {
		"tech_cultural_heritage"
	}

		building_heritage_site
	}
