building_military_academy = {
	base_buildtime = @b2_time

	empire_limit = {
		base = 1
	}

	category = army

	potential = {
		exists = owner
		owner = { NOT = { has_valid_civic = civic_reanimated_armies } }
	}

	allow = {
		has_upgraded_capital = yes
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				has_valid_civic = civic_reanimated_armies
				AND = {
					is_ai = yes
					prev = {
						NOR = {
							has_building = building_stronghold
							has_building = building_fortress
						}
					}
				}
			}
		}
	}

	convert_to = {
		building_dread_encampment
	}

	country_modifier = {
		country_commander_cap_add = 1
		commander_exp_gain = 0.33
		command_limit_add = 25
	}

	army_modifier = {
		army_starting_experience_add = 100
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				NOR = {
					has_valid_civic = civic_anarchism
					has_valid_civic = civic_feudal_realm
					has_valid_civic = civic_military_dictatorship
				}
				is_community_society = no
				is_gestalt = no
				is_primitive_society = no
				NOT = {
					has_policy_flag = military_euphony
				}
			}
		}
		modifier = {
			job_commandante_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_feudal_realm
					has_valid_civic = civic_military_dictatorship
					has_policy_flag = military_euphony
				}
				is_primitive_society = no
				is_community_society = no
			}
		}
		modifier = {
			job_knight_commander_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_anarchism
					is_community_society = yes
				}
			}
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_primitive_society = yes }
		}
		modifier = {
			job_primitive_warrior_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	prerequisites = {
		"tech_centralized_command"
	}

	show_tech_unlock_if = {
		NOT = { has_valid_civic = civic_reanimated_armies }
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
}
