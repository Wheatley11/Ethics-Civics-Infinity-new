			exists = owner

			}

		modifier = {
			planet_stability_add = 2
		}
	}
	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_conservative_armament }
		}
		starbase_shipyard_build_speed_mult = 0.005
		mult = trigger:num_pops
	}
	triggered_country_modifier = {
		potential = {
			owner = {
				has_valid_civic = civic_conservative_armament
