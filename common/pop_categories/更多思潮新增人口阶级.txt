	}

	}

		potential = {
			has_trait = trait_unnatural

		produces = {
			trigger = {
				owner = { has_valid_civic = civic_tax_inclusive_company }
			}
			energy = 2.5
		}

		upkeep = {
			energy = 1
		}
	}
