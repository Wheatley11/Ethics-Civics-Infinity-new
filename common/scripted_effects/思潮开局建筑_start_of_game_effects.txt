civic_forever_with_us_start_set = {
	optimize_memory
	create_species = {
		homeworld = this
		name = this
		namelist = this
		class = ROBOT
		portrait = default_robot
		traits = {
			trait = trait_mechanical
			trait = trait_unnatural
			trait = trait_robot_digital_1
		}
		immortal = yes
	}
	last_created_species = {
		set_citizenship_type = {
			country = prev
			type = citizenship_full_machine
		}
		set_living_standard = {
			country = prev
			type = living_standard_hive_mind
		}
		save_event_target_as = forever_species
	}
	solar_system = {
		random_system_planet = {
			limit = {
				exists = owner
				owner = {
					has_civic = civic_forever_with_us
				}
			}
			if = {
				limit = {
					owner = {
						OR = {
							has_origin = origin_advanced_start
							has_origin = origin_advanced_start
						}
					}
				}
				while = {
					count = 7
					random_owned_pop = {
						kill_pop = yes
					}
				}
				while = {
					count = 7
					create_pop = {
						species = event_target:forever_species
					}
				}
			}
			if = {
				limit = {
					owner = {
						NOR = {
							has_origin = origin_advanced_start
							has_origin = origin_advanced_start
						}
					}
				}
				while = {
					count = 3
					random_owned_pop = {
						kill_pop = yes
					}
				}
				while = {
					count = 3
					create_pop = {
						species = event_target:forever_species
					}
				}
			}
		}
	}
}
